\section{Введение}

\begin{frame}
%%\justifying 
    \frametitle{Цель диссертационной работы}
    
    Целью работы является изучение проблемы устойчивости кольцевых и линейных нейронных сетей, 
    взаимодействие в~которых описывается матричным дифферениальным урав\-не\-нием с запаздыванием $\tau$:
    \vskip-7pt
    \begin{equation}\label{maineqmatrix}
        \dot{x}(t)+A x(t)+B x(t-\tau)=0,
    \end{equation}
    где матрицы  $A, B$ содержат интенсивности мгновенного взаи\-модействия между нейронами и взаимодействия, 
    происходя\-щего с запаздыванием, соответственно.

  \vspace{0.2cm}
  Рассматривая модель \eqref{maineqmatrix},  мы имеем в виду, что
	она в общем виде описывает результат линеаризации известных
	нелинейных моделей Хопфилда-Маркуса-Вестервельта (1984,
1989) или Коэна-Гроссберга(1983) или 
Чуа--Янга--Роска (1988, 2004) вокруг некоторого решения.
  
    \vspace{0.2cm}
    Под устойчивостью уравнения мы понимаем устойчивость его нулевого решения.
\end{frame}



\begin{frame}

    \frametitle{ \centering  Задачи}
    \begin{itemize}
    %%\justifying
     \item    разработать методы анализа устойчивости уравнения 
     \eqref{maineqmatrix} произвольного порядка с запаздыванием и применить их для построения
     областей устойчивости  в пространстве параметров.
     \item выявить динамику областей устойчивости при изменении количества
     нейронов в сети и изменении запаздывания во взаимодействии нейронов;
     \item  найти области устойчивости в пространстве параметров, гарантирующие
     устойчивость независимо от величины запаздывания;
     \item  указать предельные области устойчивости, когда коли\-чество нейронов
     в кольцевой  или линейной конфигурации неограничено;
     \item  изучить переходный процесс разрыва нейронного кольца и превращения
     его в линию с точки зрения устойчивости.
      
    \end{itemize}
\end{frame}

\section{Глава 1}

\begin{frame}
%%\justifying 
    \frametitle{Глава 1. Конус устойчивости для диагностирования
устойчивости нейронных сетей. Алгоритмы и программы}

Первая глава посвящена описанию и обоснованию метода конусов
устойчивости для диагностирования устойчивости скалярного аналога уравнения
\eqref{maineqmatrix}  и собственно уравнения \eqref{maineqmatrix}  с
совместно триангулируемыми матрицами $A, B$.

  \vspace{0.3cm}
	На основе метода конуса устойчивости решена задача нахож\-дения всех значений
	запаздывания, обеспечивающих устой\-чивость уравнения \eqref{maineqmatrix} при
	фиксированных матрицах $A, B$, а также представлена программа, реализующая
	новый ана\-ли\-ти\-ческий подход к исследованию устойчивости уравнения
\eqref{maineqmatrix}.
 
    
\end{frame}
    
    
    
 \begin{frame}
    \frametitle{Конус устойчивости}
   %\begin{definition} 
   Конусом устойчивости для матричного уравнения \eqref{maineqmatrix}\\ 
    мы называем поверхность в ${\mathbb{R}}^3$, образованную
    точками $M=(u_{1}, u_{2}, u_{3})$, такими что
    \begin{equation}\label{cone}
    u_1+iu_2=(i\omega -h)\exp(-i\omega),     \qquad  u_3=h,
    \end{equation}

    где $h,\omega$ удовлетворяют неравенствам
    \begin{equation}\label{homega}
    h\geqslant -\omega{/}\tg\omega, \quad   -\pi< \omega < \pi.
    \end{equation}
% \end{definition}

\begin{frame}
    \frametitle{Основной результат главы 1:\\   критерий
    устойчивости уравнения~\eqref{maineqmatrix}}
        
    \begin{theorem}[1]\label{common:th:stab1}
    Пусть $A, B, S\in \mathbb{R}^{m\times m}$ и $S^{-1}AS=A_{T}$ и $S^{-1}BS=B_{T}$,
    где $A_{T}$ и $B_{T}$~--- нижние треугольные матрицы с~элементами соответственно 
    $\lambda_{j\,s}, \mu_{j\,s}\,(1 \leqslant  j,\, s \leqslant  m)$. 
    Построим систему точек 
    $M_j=(u_{1j},u_{2j},u_{3j})$, $(1\leqslant j \leqslant m)$,
    так, что
    \begin{equation}\label{common:pointsM}
        \begin{array}{l}
            u_{1j} = \tau \Re ( \mu_{jj} \, \exp({i\tau \Im \lambda_{jj}})),  \\
            u_{2j} = \tau \Im ( \mu_{jj} \, \exp({i\tau \Im \lambda_{jj}})),  \\
            u_{3j} = \tau \Re \lambda_{jj}.
        \end{array}
    \end{equation}
    Уравнение~\eqref{maineqmatrix} асимптотически устойчиво тогда и
    только тогда, когда все точки $M_j$ $(1\leqslant j \leqslant m)$ находятся внутри 
    конуса устойчивости. Если хотя бы одна точка 
    $M_j$  лежит вне~конуса устойчивости, то
    уравнение~\eqref{maineqmatrix} неустойчиво.
    \end{theorem}
\end{frame}

\begin{frame}
    \frametitle{Практическое исследование устойчивости системы \eqref{maineqmatrix}}
    %\justifying
    Мы сконструировали алгоритм, который анализирует принад\-лежность точек
    $M_j$ конусу устойчивости и строит систему интервалов

    \begin{center}
    $\theta = \cup_{k=1}^N (\tau_k , \, \tau_{k+1}),  $
    такую, что
    \end{center}

    \begin{itemize}
        \item $\tau\in \theta$, то система \eqref{maineqmatrix} асимптотически устойчива,
        \item $\tau\notin  [\tau_k , \, \tau_{k+1}] \quad 
        \forall k  \quad(1\leqslant k \leqslant N)$, 
        то система \eqref{maineqmatrix} неустойчива.
    \end{itemize}

%\justifying
    Полученный результат позволяет решать вопрос об устой\-чивости,
    когда значения запаздывания~$\tau$ не~являются фиксированной величиной,
    а также избежать гео\-ме\-три\-чес\-ких построений.
    
    \vspace{0.2cm}
    
    Данный алгоритм, сводящий задачу устойчивости
   \eqref{maineqmatrix}  к проверке принадлежности значе\-ния $\tau$ некоторой
   системе интервалов, положен в основу программы <<Анализ устойчивости>>.
\end{frame}

\begin{frame}
    \frametitle{Кольцевые нейронные сети}
Общим уравнением~\eqref{maineqmatrix} с совместно триангулируемыми матрицами
могут быть описаны следующие нейронные сети с несимметричным взаимодействием
нейронов с соседями
    \begin{align}
        \label{main1}
        \dot{x}_{j}(t)& +  x_{j}(t) +  a\, x_{j-1}(t) + b\, x_{j+1}(t-\tau) = 0, \\
        \label{main2}
        \dot{x}_{j}(t)& +  x_{j}(t) +  a\, x_{j-1}(t-\tau) + b\, x_{j+1}(t-\tau) = 0.
    \end{align}

      характеризуют интенсивности взаимодействия нейрона с правым и ле\-вым

\begin{frame}

    с большим количеством нейронов}
    Собственные числа матриц $A$ и $B$ для исследуемых моделей могут быть
    найдены в явном виде, что позволило найти точки $M_j^{'}$ для
    системы~\eqref{main1} и $M_j^{''}$ для системы~\eqref{main2} согласно
    вышеуказанному алгоритму и, положив $t=\frac{2\pi j}{n}$, ввести непрерывные кривые $M^{'}(t)=(u_1^{'}(t),
    u_2^{'}(t), u_3^{'}(t))$:
  \vspace{0.3cm}

    \begin{equation}\label{curve1}
    \begin{array}{l}
     u_{1}^{'}(t)+iu_{2}^{'}(t)  = \tau b \,\exp(i(-t+a\, \tau\sin t)),  \\
     u_{3}^{'}(t)             =\tau (1+a\cos t),  \quad 0\leqslant t \leqslant 2\pi
    \end{array}
    \end{equation}
    
     и $M^{''}(t)=(u_1^{''}(t), u_2^{''}(t), u_3^{''}(t))$:

    \begin{equation}\label{curve2}
    \begin{array}{l}
     u_{1}^{''}(t)+iu_{2}^{''}(t)  = \tau (a \,\exp(it) +b \,\exp(-it) ), \\
     u_{3}^{''}(t)              =\tau,   \quad 0\leqslant t \leqslant 2\pi.
    \end{array}
    \end{equation}
\end{frame}

  \begin{equation}\label{matrixbrokenequ}
\dot{x}(t) +I\,x(t)+ D\,x(t-\tau)=0,    
\end{equation}
 где  $I$ есть единичная $n\times n$ матрица, а $n\times n$ матрица $D$ имеет вид
\begin{equation}\label{matrixbroken}
D=
\begin{pmatrix}  
  0  &  b & 0 &   \ldots &  0 &  0\\
  a  &  0 & b &   \ldots &  0 &  0\\
  0  &  a & 0 &   \ldots &  0 &  0\\
  \vdots  & \vdots & \vdots  &   \ddots &  \vdots &  \vdots \\
  0  &  0 & 0 &   \ldots &  0 &  b\\
  0  &  0 & 0 &   \ldots &  a &  0   \\
\end{pmatrix}.
\end{equation}
    Введём функцию
$        F(\tau)= \dfrac{1}{4\sin^2\omega(\tau)}, \; \tau\in (0,\infty),
$
    где $\omega(\tau)$ есть наименьший положительный корень уравнения
    \begin{equation}
        \label{omegaboundary}
        \tau= \omega \tg \omega.
    \end{equation}

нейронов достаточно велико.

    Определим функцию $F1(\tau,n)$ от
запаздывания $\tau\in (0,\infty)$ и $n\in \mathbb{N}$: 
\begin{equation}
\label{F1taun} 
F1(\tau,n)= \frac{1}{4\sin^2\omega(\tau)\cos^2 \frac{\pi}{n+1} },
\end{equation}
где $\omega(\tau)$ есть наименьший положительный корень уравнения \eqref{omegaboundary}. 

    \vskip-3pt
\begin{theorem}[5]\label{common:th:stab_linear_finite}
1. Если $0\leqslant ab <\frac {1}{4\cos^2\frac{\pi}{n+1}} $, то система 
\eqref{matrixbrokenequ}, \eqref{matrixbroken} асимптотически
устойчива при любом $\tau\geqslant 0$.\\
2. Если $ab>\frac {1}{4\cos^2\frac{\pi}{n+1}} $,  то  система  \eqref{matrixbrokenequ},
\eqref{matrixbroken} неустойчива при любом $\tau\geqslant 0$. \\  
3. Если  $ab<0$ и   $|ab|<F1(\tau,n)$,     то система \eqref{matrixbrokenequ},
\eqref{matrixbroken} асимптотически устойчива.\\ 
4. Если $ab<0$ и  $|ab|>F1(\tau,n)$, то система \eqref{matrixbrokenequ},
\eqref{matrixbroken} неустойчива.  
\end{theorem}
    \vskip-3pt
Теорема 5 (ср. с Теоремой  \ref{th:stab_linear}) 
дает полное описание области устойчивости модели

   \begin{frame}

    \frametitle{Моделирование разрыва кольцевой сети нейронов}
    
Пусть в модели кольцевой сети
из шести нейронов интенсивности взаимодействия между всеми нейронами, кроме
первого и шестого, равны $a$ и $b$ (как в модели~\eqref{main2}), а  интенсивности воздействия 
шестого нейрона на первый и первого нейрона на шестой равны соответственно 
$ac$ и $bc$, где $c\geq 0$ -- некоторый параметр.


Тогда взаимодействие нейронов в описанной модели  задается
уравнением~\eqref{maineqmatrix} с~матрицами
\begin{equation}\label{common:matrixbroken_c}
A=I, \quad
B=
\begin{pmatrix}  
  0  &  b & 0 &   0 &  0 &  ac\\[-1mm]
  a  &  0 & b &  0 &  0 &  0\\[-1mm]
  0  &  a & 0 &   b &  0 &  0\\[-1mm]
  0  & 0 & a  &   0 &  b &  0 \\[-1mm]
  0  &  0 & 0 &   a &  0 &  b\\[-1mm]
  bc  &  0 & 0 &   0 &  a &  0   \\
\end{pmatrix}.
\end{equation}
При $c=1$ система~\eqref{maineqmatrix},~\eqref{common:matrixbroken_c}
описывает кольцевую сеть~\eqref{main2}.
С изменением $c$ от единицы до нуля интенсивность взаимодействия между первым 
и шестым нейронами в системе~\eqref{maineqmatrix},~\eqref{common:matrixbroken_c} 
постепенно ослабевает, и при $c=0$ кольцо нейронов размыкается. 
      \end{frame}
