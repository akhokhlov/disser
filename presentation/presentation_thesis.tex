\section{Введение}

\begin{frame}
%%\justifying 
    \frametitle{Цель диссертационной работы}
    
    Целью работы является изучение проблемы устойчивости кольцевых и линейных нейронных сетей, 
    взаимодействие в~которых описывается матричным дифферениальным урав\-не\-нием с запаздыванием $\tau$:
    
    \begin{equation}\label{maineqmatrix}
        \dot{x}(t)+A x(t)+B x(t-\tau)=0,
    \end{equation}
    где матрицы  $A, B$ содержат интенсивности мгновенного взаи\-модействия между нейронами и взаимодействия, 
    происходя\-щего с запаздыванием, соответственно.

  \vspace{0.3cm}
  
    Под устойчивостью уравнения мы понимаем устойчивость его нулевого решения.
\end{frame}



\begin{frame}

    \frametitle{Задачи}
    \begin{itemize}
    %%\justifying
     \item    разработать методы анализа устойчивости систем \eqref{maineqmatrix}  
     произвольного порядка с запаздыванием и применить их для построения
     областей устойчивости  в пространстве параметров.
     \item выявить динамику областей устойчивости при изменении количества
     нейронов в сети и изменении запаздывания во взаимодействии нейронов;
     \item  найти области устойчивости в пространстве параметров, гарантирующие
     устойчивость независимо от величины запаздывания;
     \item  указать предельные области устойчивости, когда коли\-чество нейронов
     в кольцевой  или линейной конфигурации неограничено;
     \item  изучить переходный процесс разрыва нейронного кольца и превращения
     его в линию с точки зрения устойчивости.
      
    \end{itemize}
\end{frame}

\begin{frame}
%%\justifying 
    \frametitle{Методы исследования}
    \begin{itemize}
    %%\justifying
     \item Поставленные задачи решаются в диссертации методом конуса
     устойчивости, разработанным  автором совместно с научным руководителем и
     В.В. Малыгиной.
	\item  На основе метода в диссертации построены алгоритмы для поиска значений
 	запаздываний, гарантирующих ус\-тойчивость системы при фиксированных
	коэффициентах.
	\item В свою очередь, алгоритмы реализованы в виде программ для анализа
	устойчивости как для общих систем, так и для  специальных систем, описывающих
	кольцевые и линейные нейронные сети с запаздываниями во взаимо\-действии
	соседних нейронов.
    \item Отдельно разработаны алгоритм и программа для пост\-роения областей
    устойчивости исследуемых нейронных сетей в плоскости параметров.
    \end{itemize}
\end{frame}


\section{Глава 1}

\begin{frame}
%%\justifying 
    \frametitle{Глава 1. Конус устойчивости для диагностирования
устойчивости нейронных сетей. Алгоритмы и программы}

Первая глава посвящена описанию и обоснованию метода конусов
устойчивости для диагностирования устойчивости скалярного аналога уравнения
\eqref{maineqmatrix}  и собственно уравнения \eqref{maineqmatrix}  с
совместно триангулируемыми матрицами $A, B$.

  \vspace{0.3cm}
	На основе метода конуса устойчивости решена задача нахож\-дения всех значений
	запаздывания, обеспечивающих устой\-чивость уравнения \eqref{maineqmatrix} при
	фиксированных матрицах $A, B$, а также представлена программа, реализующая
	новый ана\-ли\-ти\-ческий подход к исследованию устойчивости уравнения
\eqref{maineqmatrix}.
 
    
\end{frame}
    
    
    
 \begin{frame}
    \frametitle{Конус устойчивости}
   %\begin{definition} 
   Конусом устойчивости для матричного уравнения \eqref{maineqmatrix}\\ 
    мы называем поверхность в ${\mathbb{R}}^3$, образованную
    точками $M=(u_{1}, u_{2}, u_{3})$, такими что
    \begin{equation}\label{cone}
    u_1+iu_2=(i\omega -h)\exp(-i\omega),     \qquad  u_3=h,
    \end{equation}

    где $h,\omega$ удовлетворяют неравенствам
    \begin{equation}\label{homega}
    h\geqslant -\omega{/}\tg\omega, \quad   -\pi< \omega < \pi.
    \end{equation}
% \end{definition}

\begin{frame}
    \frametitle{Основной результат главы 1: критерий устойчивости уравнения~\eqref{maineqmatrix}}
        
    \begin{theorem}[1]\label{common:th:stab1}
    Пусть $A, B, S\in \mathbb{R}^{m\times m}$ и $S^{-1}AS=A_{T}$ и $S^{-1}BS=B_{T}$,
    где $A_{T}$ и $B_{T}$~--- нижние треугольные матрицы с~элементами соответственно 
    $\lambda_{j\,s}, \mu_{j\,s}\,(1 \leqslant  j,\, s \leqslant  m)$. 
    Построим систему точек 
    $M_j=(u_{1j},u_{2j},u_{3j})$, $(1\leqslant j \leqslant m)$,
    так, что
    \begin{equation}\label{common:pointsM}
        \begin{array}{l}
            u_{1j} = \tau \Re ( \mu_{jj} \, \exp({i\tau \Im \lambda_{jj}})),  \\
            u_{2j} = \tau \Im ( \mu_{jj} \, \exp({i\tau \Im \lambda_{jj}})),  \\
            u_{3j} = \tau \Re \lambda_{jj}.
        \end{array}
    \end{equation}
    Уравнение~\eqref{maineqmatrix} асимптотически устойчиво тогда и
    только тогда, когда все точки $M_j$ $(1\leqslant j \leqslant m)$ находятся внутри 
    конуса устойчивости. Если хотя бы одна точка 
    $M_j$  лежит вне~конуса устойчивости, то
    уравнение~\eqref{maineqmatrix} неустойчиво.
    \end{theorem}
\end{frame}

\begin{frame}
    \frametitle{Практическое исследование устойчивости системы \eqref{maineqmatrix}}
    %\justifying
    Мы сконструировали алгоритм, который анализирует принад\-лежность точек
    $M_j$ конусу устойчивости и строит систему интервалов

    \begin{center}
    $\theta = \cup_{k=1}^N (\tau_k , \, \tau_{k+1}),  $
    такую, что
    \end{center}

    \begin{itemize}
        \item $\tau\in \theta$, то система \eqref{maineqmatrix} асимптотически устойчива,
        \item $\tau\notin  [\tau_k , \, \tau_{k+1}] \quad 
        \forall k  \quad(1\leqslant k \leqslant N)$, 
        то \eqref{maineqmatrix} неустойчива.
    \end{itemize}

%\justifying
    Полученный результат позволяет решать вопрос об устой\-чивости,
    когда значения запаздывания~$\tau$ не~являются фиксированной величиной,
    а также избежать гео\-ме\-три\-чес\-ких построений.
    Алгоритм положен в основу программы <<Анализ устойчивости>>.
\end{frame}

\begin{frame}
    \frametitle{Круговые нейронные сети}
Общим уравнением~\eqref{maineqmatrix} с совместно триангулируемыми матрицами
могут быть описаны следующие нейронные сети с несимметричным взаимодействием
нейронов с соседями
    \begin{align}
        \label{main1}
        \dot{x}_{j}(t)& +  x_{j}(t) +  a\, x_{j-1}(t) + b\, x_{j+1}(t-\tau) = 0, \\
        \label{main2}
        \dot{x}_{j}(t)& +  x_{j}(t) +  a\, x_{j-1}(t-\tau) + b\, x_{j+1}(t-\tau) = 0.
    \end{align}

\begin{tabular}[c]{@{}p{7.5cm}@{}@{}p{4cm}@{}}
\begin{itemize}
      %\justifying
      \item сигнал $j$-го нейрона в момент времени $t$ описывается переменной
      \bluetext{$x_j(t)$} $(1\leqslant j \leqslant n)$,
      \item действительные коэффициенты \bluetext{$a$} и \bluetext{$b$}
      характеризуют интенсивности взаимодействия нейрона с правым и ле\-вым
      соседними нейронами.
    \end{itemize}

\end{frame}



\begin{frame}
    \frametitle{Модификация алгоритма для систем\\ 
    с большим количеством нейронов}
    Собственные числа матриц $A$ и $B$ для системы~\eqref{main1} 
    \begin{equation}
    \label{lambdamu1}
    \lambda_{j}^{'} = 1+a\,\exp(i\frac{2\pi j}{n}), \,  
    \mu_{j}^{'}= b\,\exp(-i\frac{2\pi j}{n}), \, 1\leqslant j \leqslant n.
    \end{equation}

    Собственные числа матриц $A$ и $B$ для системы~\eqref{main2} 

    \begin{equation}
    \label{lambdamu2}
    \lambda_{j}^{''} = 1, \,   \mu_{j}^{''}=a\, \exp(i\frac{2\pi j}{n}) + b\, 
    \exp(-i\frac{2\pi j}{n}), \, 1\leqslant j \leqslant n.
    \end{equation}
\end{frame}



\begin{frame}
    \frametitle{Модификация алгоритма для систем\\ 
    с большим количеством нейронов}
    Согласно вышеуказанному алгоритму найдём $M_j^{'}$ для \eqref{main1} и 
    $M_j^{''}$ для \eqref{main2} и, положив  $t=\frac{2\pi j}{n}$, введём
    непрерывные кривые $M^{'}(t)=(u_1^{'}(t), u_2^{'}(t), u_3^{'}(t))$:
  \vspace{0.3cm}

    \begin{equation}\label{curve1}
    \begin{array}{l}
     u_{1}^{'}(t)+iu_{2}^{'}(t)  = \tau b \,\exp(i(-t+a\, \tau\sin t)),  \\
     u_{3}^{'}(t)             =\tau (1+a\cos t),  \quad 0\leqslant t \leqslant 2\pi.
    \end{array}
    \end{equation}

  \vspace{0.3cm}

    и $M^{''}(t)=(u_1^{''}(t), u_2^{''}(t), u_3^{''}(t))$:

    \begin{equation}\label{curve2}
    \begin{array}{l}
     u_{1}^{''}(t)+iu_{2}^{''}(t)  = \tau (a \,\exp(it) +b \,\exp(-it) ), \\
     u_{3}^{''}(t)              =\tau,   \quad 0\leqslant t \leqslant 2\pi.
    \end{array}
    \end{equation}
\end{frame}


\begin{frame}
%\justifying
    \frametitle{Критерий диагностирования устойчивости \\
    для сетей с большим количеством нейронов}
Если все точки кривой $M^{'}(t)$ лежат внутри конуса устой\-чи\-вости, 
то уравнение \eqref{main1} \alert{асимптотически устойчиво} при любом~$n$,  
а если хотя~бы одна точка $M^{'}(t)$ кривой лежит вне конуса устойчивости,  
то уравнение \eqref{main1} \alert{неустойчиво} при всех достаточно больших
значениях $n$.
Аналогично изучается уравнение \eqref{main2}.

\begin{frame}
    \frametitle{Линейная система нейронов}
%\justifying
    \begin{figure}[htp]
        \centering
        \includegraphics[width=6cm]{line_of_neurons}
    \end{figure}
    \begin{equation}\label{main3}
        \left\{
        \begin{array}{l}
            \dot{x}_{1}(t) +  x_{1}(t) +   b\, x_{2}(t-\tau) = 0,  \\ \\
            \dot{x}_{j}(t) +  x_{j}(t) +  a\, x_{j-1}(t-\tau) + b\, x_{j+1}(t-\tau) = 0, \\  
            2\leqslant j\leqslant (n-1), \\ \\
            \dot{x}_{n}(t) +  x_{n}(t) +  a\, x_{n-1}(t-\tau)  = 0, \qquad n>2.
        \end{array}
        \right.
    \end{equation}

    Введём функцию
    \begin{equation}
        \label{Ftau}
        F(\tau)= \frac{1}{4\sin^2\omega(\tau)}, \qquad \tau\in (0,\infty),
    \end{equation}
    где $\omega(\tau)$ есть наименьший положительный корень уравнения
    \begin{equation}
        \label{omegaboundary}
        \tau= \omega \tg \omega.
    \end{equation}
\end{frame}



\begin{frame}
    \frametitle{Теоретические результаты}

    \begin{theorem}[3] \label{th2}
    \begin{enumerate}

        \item Если  $|ab|<\frac {1}{4} $, то система  \eqref{main3} 
        \bluetext{асимптотически ус\-той\-чива} \\при любом $n$ и любом $\tau\geqslant 0$.
        \vspace{0.1cm}
        \item Если $ab> \frac {1}{4}$, то  система  \eqref{main3} \bluetext{неустойчива} 
        при любом $\tau\geqslant 0$, \\
        если $n$ достаточно велико.
               \vspace{0.1cm}
        \item Если  $ab<0$ и $|ab|<F(\tau)$, то система \eqref{main3} 
        \bluetext{асимп\-тотически устойчива}  при лю\-бом $n$.
               \vspace{0.1cm}
        \item Если $ab<0$ и  $|ab|>F(\tau)$, то система \eqref{main3} 
        \bluetext{неус\-той\-чива}, \\
        если $n$ достаточно велико.
    \end{enumerate}
    \end{theorem}
    Теорема 3 позволяет полностью описать области устой\-чи\-вости и неустойчивости
   системы  \eqref{main3} в плоскости параметров. 
\end{frame}

\begin{frame}
	\frametitle{Сравнение областей устойчивости кольцевой ~\eqref{main2} и линейной 
	\eqref{main3} сетей нейронов с неограниченным числом нейронов}
	
	\begin{theorem}[4]\label{common:th:comparison_ring_and_linear} 
%\justifying 
		Фиксируем значения параметров $a,b, \tau$ для системы~\eqref{main2}, 
		 для кольца нейронов и  системы ~\eqref{main3}  для сети с линейной
		 конфигурацией нейронов.
		Если система~\eqref{main2}  асимптотически устойчива при любом количестве
		нейронов $n$, то система~\eqref{main3} также асимптотически устойчива при
		любом   количестве нейронов $n$.
	\end{theorem}
	
	\vspace{5mm}
	Теорема \ref{common:th:comparison_ring_and_linear} дает твердые основания к
	заключению, что разрыв кольца нейронов расширяет область устойчивости сети при
	большом количестве нейронов в сети.
\end{frame}
