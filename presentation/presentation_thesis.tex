\section{Введение}

\begin{frame}
%%\justifying 
    \frametitle{Цель диссертационной работы}
    
    Целью работы является изучение проблемы устойчивости кольцевых и линейных нейронных сетей, 
    взаимодействие в~которых описывается матричным дифференциальным
    урав\-не\-нием с запаздыванием $\tau$:
    \vskip-7pt
    \begin{equation}\label{maineqmatrix}
        \dot{x}(t)+A x(t)+B x(t-\tau)=0.
    \end{equation}
    %где матрицы  $A, B$ содержат интенсивности мгновенного взаи\-модействия
    % между нейронами и взаимодействия, происходя\-щего с запаздыванием, соответственно.

  \vspace{0.2cm}
 Данная модель в общем виде описывает результат
 линеаризации известной нелинейной модели Хопфилда-Маркуса-Вестервельта (1984,
1989) 
\begin{equation*}\label{delayednonlinearmodel}
    C_j\dot{x}_j(t)+\frac{1}{R_j}x_j(t)+\sum_{k=1}^n T_{jk}g_k (x_k(t-\tau))=0 , 
    \quad j=1,2,\ldots,n,     
\end{equation*}

или модели Коэна-Гроссберга(1983) или модели Чуа--Янга--Роска (1988, 2004)
вокруг некоторого решения.
  
    \vspace{0.2cm}
    Под устойчивостью уравнения (1) мы понимаем устойчивость его нулевого
    решения.
\end{frame}



\begin{frame}

    \frametitle{ \centering  Задачи}
    \begin{itemize}
    %%\justifying
     \item    разработать методы анализа устойчивости уравнения  $\dot{x}(t)+A
     x(t)+B x(t-\tau)=0$ произвольного порядка с запаздыванием и применить их для построения
     областей устойчивости  в пространстве параметров;
     \item выявить динамику областей устойчивости при изменении количества
     нейронов в сети и изменении запаздывания во взаимодействии нейронов;
     \item  найти области устойчивости в пространстве параметров, гарантирующие
     устойчивость независимо от величины запаздывания;
     \item  указать предельные области устойчивости, когда коли\-чество нейронов
     в кольцевой  или линейной конфигурации неограничено;
     \item  изучить переходный процесс разрыва нейронного кольца и превращения
     его в линию с точки зрения устойчивости.
        
         \end{itemize}
        
\end{frame}


\begin{frame}
 \frametitle{ \centering  Методы исследования}
    \begin{itemize}

     \item Поставленные задачи решаются в диссертации методом конуса
     устойчивости, разработанным  автором совместно с научным руководителем и
     В.В. Малыгиной.\\
     \vskip 2pt 
     Идейно конус устойчивости происходит от давней работы З.~Рехлицкого (1956),
     в которой был построен овал устойчивости для некоторого общего
     уравнения вида $\dot{x}(t)+B x(t-\tau)=0$.
     
     \item На основе данного метода построены алгоритмы для
     поиска значений запаздываний, гарантирующих ус\-тойчивость системы при фиксированных
	коэффициентах.
	\item Алгоритмы реализованы в виде программ (разработанных в пакете Matlab 
	7.11.0) для анализа устойчивости как общих систем, так и специальных систем, описывающих
	кольцевые и линейные нейронные сети. 
 \end{itemize}
 

\end{frame}

\begin{frame}
%%\justifying 
    \frametitle{Глава 1. Конус устойчивости для диагностирования
устой\-чивости нейронных сетей. Алгоритмы и программы}

     Определение. \bluetext{Конусом устойчивости} для матричного уравнения
     $\dot{x}(t)+A x(t)+B x(t-\tau)=0$
    мы называем поверхность в ${\mathbb{R}}^3$, образованную
    точками $M=(u_{1}, u_{2}, u_{3})$, такими что
     \vskip-11pt
    \begin{equation*}\label{cone}
    u_1+iu_2=(i\omega -h)\exp(-i\omega),     \qquad  u_3=h,
    \end{equation*}
  \vskip-1pt
    где $h,\omega$ удовлетворяют неравенствам
     \vskip-11pt
    \begin{equation*}\label{homega}
    h\geqslant -\omega{/}\tg\omega, \quad   -\pi< \omega < \pi.
    \end{equation*}

\begin{frame}
    \frametitle{Основной результат главы 1:  критерий\\
    устойчивости уравнения $\large \dot{x}(t)+A x(t)+B x(t-\tau)=0\quad (1)$}
        
    \begin{theorem}\label{common:th:stab1}
    Пусть $A, B, S\in \mathbb{R}^{m\times m}$ и $S^{-1}AS=A_{T}$ и $S^{-1}BS=B_{T}$,
    где $A_{T}$ и $B_{T}$~--- нижние треугольные матрицы с~элементами соответственно 
    $\lambda_{j\,s}, \mu_{j\,s}\,(1 \leqslant  j,\, s \leqslant  m)$. 
    Построим систему точек 
    $M_j=(u_{1j},u_{2j},u_{3j})$, $(1\leqslant j \leqslant m)$,
    так, что
    \begin{equation*}\label{common:pointsM}
        \begin{array}{l}
            u_{1j} = \tau \Re ( \mu_{jj} \, \exp({i\tau \Im \lambda_{jj}})),  \\
            u_{2j} = \tau \Im ( \mu_{jj} \, \exp({i\tau \Im \lambda_{jj}})),  \\
            u_{3j} = \tau \Re \lambda_{jj}.
        \end{array}
    \end{equation*}
    Уравнение (1) асимптотически устойчиво тогда и
    только тогда, когда все точки $M_j$ $(1\leqslant j \leqslant m)$ находятся внутри 
    конуса устойчивости. Если хотя бы одна точка 
    $M_j$  лежит вне~конуса устойчивости, то
    уравнение (1) неустойчиво.
    \end{theorem}
\end{frame}

\section{Глава 2}
\begin{frame}
%\justifying 
    \frametitle{Глава 2. Численный и теоретический анализ устойчивости
    моделей нейронных сетей кольцевой и линейной конфигураций с неограниченным
    количеством нейронов }

   Общим уравнением  $\dot{x}(t)+A x(t)+B x(t-\tau)=0$ с совместно
   триангулируемыми матрицами описываются следующие кольцевые сети с
несимметричным взаимодействием нейронов с соседями 
\bluetext{c~односторонним запаздыванием}
  
  \vskip-10pt
        \begin{equation}
        \label{main1}
     \dot{x}_{j}(t)& + 
        x_{j}(t) +  a\, x_{j-1}(t) + b\, x_{j+1}(t-\tau) = 0  \quad (j\;
        {\rm{mod}}\; n)
        \end{equation}
 
  и \bluetext{c~двусторонним запаздыванием}       
\begin{equation}\label{main2}
    \dot{x}_{j}(t)& +  x_{j}(t) +  a\, x_{j-1}(t-\tau) + b\, x_{j+1}(t-\tau) = 0
    \quad (j\; {\rm{mod}}\; n).
\end{equation}
   \vskip-10pt
\begin{tabular}[c]{@{}p{6.5cm}@{}@{}p{5cm}@{}}
      \justifying
      \vskip2pt
   \bluetext{$x_j(t)$} есть сигнал $j$-го нейрона  в момент  $t$,
   \bluetext{$a$}~и~\bluetext{$b$} -- интенсивности взаимодействия нейрона с
   правым и ле\-вым соседними нейронами.

\begin{frame}
    \frametitle{Модификация алгоритма для систем~\eqref{main1},~\eqref{main2}\\
    с большим количеством нейронов}
    Собственные числа матриц $A$ и $B$ для исследуемых моделей могут быть
    найдены в явном виде, что позволило найти точки $M_j^{'}$ для
    системы~\eqref{main1} и $M_j^{''}$ для системы~\eqref{main2} согласно
    вышеуказанному алгоритму и, положив $s=\frac{2\pi j}{n}$, ввести непрерывные кривые $M^{'}(s)=(u_1^{'}(s),
    u_2^{'}(s), u_3^{'}(s))$:
  \vspace{0.3cm}

    \begin{equation*}\label{curve1}
    \begin{array}{l}
     u_{1}^{'}(s)+iu_{2}^{'}(s)  = \tau b \,\exp(i(-s+a\, \tau\sin s)),  \\
     u_{3}^{'}(s)             =\tau (1+a\cos s),  \quad 0\leqslant s \leqslant 2\pi
    \end{array}
    \end{equation*}
    
     и $M^{''}(s)=(u_1^{''}(s), u_2^{''}(s), u_3^{''}(s))$:

    \begin{equation*}\label{curve2}
    \begin{array}{l}
     u_{1}^{''}(s)+iu_{2}^{''}(s)  = \tau (a \,\exp(is) +b \,\exp(-is) ), \\
     u_{3}^{''}(s)              =\tau,   \quad 0\leqslant s \leqslant 2\pi.
    \end{array}
    \end{equation*}
\end{frame}

\begin{frame}
%\justifying
    \frametitle{Критерий устойчивости
    для сетей \\с большим количеством нейронов}
Если все точки кривой $M^{'}(s)$ лежат внутри конуса устой\-чи\-вости, 
то уравнение \eqref{main1}  \bluetext{асимптотически устойчиво} при любом~$n$,  
а если хотя~бы одна точка $M^{'}(s)$ кривой лежит вне конуса устойчивости,  
то уравнение \eqref{main1}  \bluetext{неустойчиво} при всех достаточно больших
значениях $n$.
Аналогично изучается уравнение \eqref{main2}.

    \frametitle{Моделирование разрыва кольцевой сети нейронов}
    
Пусть в модели кольцевой сети
из шести нейронов интенсивности взаимодействия между всеми нейронами, кроме
первого и шестого, равны $a$ и $b$ (как в модели~\eqref{main2}), а  интенсивности воздействия 
шестого нейрона на первый и первого нейрона на шестой равны соответственно 
$ac$ и $bc$, где $c\geq 0$ -- некоторый параметр.


Тогда взаимодействие нейронов в описанной модели  задается
уравнением  $\dot{x}(t)+A x(t)+B x(t-\tau)=0$ с~матрицами
\begin{equation*}\label{common:matrixbroken_c}
A=I, \quad
B=
\begin{pmatrix}  
  0  &  b & 0 &   0 &  0 &  ac\\[-1mm]
  a  &  0 & b &  0 &  0 &  0\\[-1mm]
  0  &  a & 0 &   b &  0 &  0\\[-1mm]
  0  & 0 & a  &   0 &  b &  0 \\[-1mm]
  0  &  0 & 0 &   a &  0 &  b\\[-1mm]
  bc  &  0 & 0 &   0 &  a &  0   \\
\end{pmatrix}.
\end{equation*}
При $c=1$ данная система
описывает кольцевую сеть~\eqref{main2}.
С изменением $c$ от единицы до нуля интенсивность взаимодействия между первым 
и шестым нейронами 
постепенно ослабевает, и при $c=0$ кольцо нейронов размыкается (образуется сеть 
\eqref{matrixbrokenequ}, \eqref{matrixbroken}).

Реакции нейрона на сигнал правого и левого соседа обладают  одинаковыми
интенсивностями $\beta$ и запаздыванием $\tau$. Сложность  динамики
происходит из-за искусственного разделения реакции нейрона на собственный сигнал на две составляющие:
мгновенную с интенсивностью 1 и запаздывающую на время $\tau_s$ с интенсивностью $\alpha$,
что свидетельствует о недостаточной адекватности модели.

В нашей модели источником сложности поведения модели является несимметричность
(когда $a\neq b$) и даже антисимметричность (когда $a=-b$) реакции нейрона на сигналы правого и левого
соседа, что соответствует традиционному разделению сигналов на возбуждающие и
тормозящие.  


В диссертации решена задача исследования устойчивости общего
уравнения $\dot{x}(t)+Ax(t)+B x(t-\tau)=0$ и его частных случаев в зависимости от
коэффициентов моделей, величины запаздывания и количества нейронов в сети.
